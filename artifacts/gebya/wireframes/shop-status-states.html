Health Algorithm

computeShopStory(metrics, priorMetrics, creditHealth, closingDone, cashVariance) {
  let score = 4;

  // Factor 1: Profit
  if (metrics.totalSold - metrics.spentToday <= 0) score--;

  // Factor 2: Cash match (after closing)
  if (closingDone && Math.abs(cashVariance) > metrics.cashExpected * 0.05) score--;

  // Factor 3: Credit health
  if (creditHealth.overdueRatio > 0.3) score--;

  // Factor 4: Sales vs average
  if (priorMetrics && metrics.totalSold < priorMetrics.totalSold * 0.5) score--;

  // Determine status
  if (score >= 4) return { status: 'healthy', emoji: '🟢', headline: 'Business is healthy today' };
  if (score >= 3) return { status: 'warning', emoji: '🟡', headline: 'Needs attention' };
  return { status: 'urgent', emoji: '🔴', headline: 'Urgent issues found' };
}
